#+begin_src python

import matplotlib
matplotlib.use('Agg')
import matplotlib.pyplot as plt
import numpy as np

# First plot: Sine wave
plt.figure(figsize=(10, 6))
x = np.linspace(0, 4*np.pi, 200)
y = np.sin(x)
plt.plot(x, y, 'b-', label='sin(x)')
plt.grid(True)
plt.title('Sine Wave')
plt.xlabel('x')
plt.ylabel('y')
plt.legend()
plt.savefig('results/figures/sine.jpg', bbox_inches='tight', dpi=300)
plt.close()

# Second plot: Sine and Cosine
plt.figure(figsize=(10, 6))
y_cos = np.cos(x)
plt.plot(x, y, 'b-', label='sin(x)')
plt.plot(x, y_cos, 'r--', label='cos(x)')
plt.grid(True)
plt.title('Sine and Cosine Waves')
plt.xlabel('x')
plt.ylabel('y')
plt.legend()
plt.savefig('results/figures/sine_cosine.jpg', bbox_inches='tight', dpi=300)
plt.close()
